            osc.type = 'sawtooth';

            osc.frequency.exponentialRampToValueAtTime(100, audioCtx.currentTime + 1.5);
            gain.gain.setValueAtTime(0.3, audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 1.5);

        function getPath(sx, sy, gx, gy) {
            let q = [{x: sx, y: sy, path: []}];
            let vis = Array.from({length: MAP_SIZE}, () => Array(MAP_SIZE).fill(false));
            vis[sy][sx] = true;
            
            let dirs = [[0,-1], [0,1], [-1,0], [1,0]];
            while (q.length > 0) {
                let cur = q.shift();
                if (cur.x === gx && cur.y === gy) return cur.path;
                
                for (let d of dirs) {
                    let nx = cur.x + d[0], ny = cur.y + d[1];
                    if (nx >= 0 && nx < MAP_SIZE && ny >= 0 && ny < MAP_SIZE && worldMap[ny][nx] !== 1 && !vis[ny][nx]) {
                        vis[ny][nx] = true;
                        q.push({x: nx, y: ny, path: [...cur.path, {x: nx + 0.5, y: ny + 0.5}]});
                    }
                }
            }
            return [];
        }

        function render3D() {
            let grad = ctx.createLinearGradient(0, 0, 0, SCREEN_H);
            grad.addColorStop(0, '#010101'); 
            grad.addColorStop(0.5, '#000'); 
            grad.addColorStop(1, '#080000'); 
            ctx.fillStyle = grad;
            ctx.fillRect(0, 0, SCREEN_W, SCREEN_H);

            let camZ = 0.5 - ((player.maxHp - player.hp) * 0.1); 

            for (let x = 0; x < SCREEN_W; x++) {
                let cameraX = 2 * x / SCREEN_W - 1;
                let rayDirX = player.dirX + player.planeX * cameraX;
                let rayDirY = player.dirY + player.planeY * cameraX;

                let mapX = Math.floor(player.x);
                let mapY = Math.floor(player.y);

                let sideDistX, sideDistY;
                let deltaDistX = Math.abs(1 / rayDirX);
                let deltaDistY = Math.abs(1 / rayDirY);
                let perpWallDist, stepX, stepY, hit = 0, side;

                if (rayDirX < 0) { stepX = -1; sideDistX = (player.x - mapX) * deltaDistX; } 
                else { stepX = 1; sideDistX = (mapX + 1.0 - player.x) * deltaDistX; }
                
                if (rayDirY < 0) { stepY = -1; sideDistY = (player.y - mapY) * deltaDistY; } 
                else { stepY = 1; sideDistY = (mapY + 1.0 - player.y) * deltaDistY; }

                while (hit === 0) {
                    if (sideDistX < sideDistY) { sideDistX += deltaDistX; mapX += stepX; side = 0; } 
                    else { sideDistY += deltaDistY; mapY += stepY; side = 1; }
                    if (worldMap[mapY][mapX] > 0) hit = 1;
                }

                if (side === 0) perpWallDist = (mapX - player.x + (1 - stepX) / 2) / rayDirX;
                else perpWallDist = (mapY - player.y + (1 - stepY) / 2) / rayDirY;

                zBuffer[x] = perpWallDist; 

                let lineHeight = Math.floor(SCREEN_H / perpWallDist);
                let drawStart = Math.floor(-lineHeight * (1 - camZ) + SCREEN_H / 2);
                let drawEnd = Math.floor(lineHeight * camZ + SCREEN_H / 2);

                let color = (worldMap[mapY][mapX] === 1) ? 
                            (side === 1 ? [20, 20, 20] : [10, 10, 10]) : 
                            (side === 1 ? [150, 0, 0] : [100, 0, 0]);

                let fog = Math.max(0, 1 - (perpWallDist / 6)); 
                if (worldMap[mapY][mapX] >= 2) fog = Math.max(0.1, 1 - (perpWallDist / 10));

                ctx.fillStyle = `rgb(${color[0]*fog}, ${color[1]*fog}, ${color[2]*fog})`;
                ctx.fillRect(x, drawStart, 1, drawEnd - drawStart);
            }

            if (enemy.active) {
                let spriteX = enemy.x - player.x;
                let spriteY = enemy.y - player.y;

                let invDet = 1.0 / (player.planeX * player.dirY - player.dirX * player.planeY);
                let transformX = invDet * (player.dirY * spriteX - player.dirX * spriteY);
                let transformY = invDet * (-player.planeY * spriteX + player.planeX * spriteY);

                if (transformY > 0.1) {
                    let spriteScreenX = Math.floor((SCREEN_W / 2) * (1 + transformX / transformY));
                    let spriteBottom = Math.floor((SCREEN_H / transformY) * camZ + SCREEN_H / 2);
                    
                    let enemyScale = 0.85; 
                    let spriteHeight = Math.abs(SCREEN_H / transformY) * enemyScale;
                    let spriteWidth = spriteHeight / 2;

                    if (spriteScreenX > -spriteWidth && spriteScreenX < SCREEN_W + spriteWidth && transformY < zBuffer[spriteScreenX]) {
                        drawEnemySprite(spriteScreenX, spriteBottom, spriteHeight, enemy.stunTimer > 0);
                    }
                }
            }
        }

        function updateGame() {
            if (isGamePaused) return;

            if (player.hitCooldown > 0) player.hitCooldown--;
            if (stageCooldown > 0) stageCooldown--;

            const moveSpeed = 0.05;
            const rotSpeed = 0.04;
            const collisionBuffer = 1.2; 

            if (keys.w) {
                if (worldMap[Math.floor(player.y)][Math.floor(player.x + player.dirX * moveSpeed * collisionBuffer)] === 0) player.x += player.dirX * moveSpeed;
                if (worldMap[Math.floor(player.y + player.dirY * moveSpeed * collisionBuffer)][Math.floor(player.x)] === 0) player.y += player.dirY * moveSpeed;
            }
            if (keys.s) {
                if (worldMap[Math.floor(player.y)][Math.floor(player.x - player.dirX * moveSpeed * collisionBuffer)] === 0) player.x -= player.dirX * moveSpeed;
                if (worldMap[Math.floor(player.y - player.dirY * moveSpeed * collisionBuffer)][Math.floor(player.x)] === 0) player.y -= player.dirY * moveSpeed;
            }
            if (keys.d) {
                let oldDirX = player.dirX;
                player.dirX = player.dirX * Math.cos(rotSpeed) - player.dirY * Math.sin(rotSpeed);
                player.dirY = oldDirX * Math.sin(rotSpeed) + player.dirY * Math.cos(rotSpeed);
                let oldPlaneX = player.planeX;
                player.planeX = player.planeX * Math.cos(rotSpeed) - player.planeY * Math.sin(rotSpeed);
                player.planeY = oldPlaneX * Math.sin(rotSpeed) + player.planeY * Math.cos(rotSpeed);
            }
            if (keys.a) {
                let oldDirX = player.dirX;
                player.dirX = player.dirX * Math.cos(-rotSpeed) - player.dirY * Math.sin(-rotSpeed);
                player.dirY = oldDirX * Math.sin(-rotSpeed) + player.dirY * Math.cos(-rotSpeed);
                let oldPlaneX = player.planeX;
                player.planeX = player.planeX * Math.cos(-rotSpeed) - player.planeY * Math.sin(-rotSpeed);
                player.planeY = oldPlaneX * Math.sin(-rotSpeed) + player.planeY * Math.cos(-rotSpeed);
            }

            if (stageCooldown === 0) {
                checkStageCollision();
            }

            if (enemy.active) {
                if (enemy.stunTimer > 0) {
                    enemy.stunTimer--;
                } else {
                    let dx = player.x - enemy.x;
                    let dy = player.y - enemy.y;
                    let dist = Math.sqrt(dx*dx + dy*dy);
                    
                    let moveSpeed = 0.015; 

                    if (dist > 0.4) {
                        enemy.pathTimer++;
                        if (enemy.pathTimer > 20 || !enemy.path || enemy.path.length === 0) {
                            enemy.path = getPath(Math.floor(enemy.x), Math.floor(enemy.y), Math.floor(player.x), Math.floor(player.y));
                            enemy.pathTimer = 0;
                        }

                        if (enemy.path && enemy.path.length > 0) {
                            let tx = enemy.path[0].x;
                            let ty = enemy.path[0].y;
                            let tdx = tx - enemy.x;
                            let tdy = ty - enemy.y;
                            let tdist = Math.sqrt(tdx*tdx + tdy*tdy);
                            
                            if (tdist < 0.2) {
                                enemy.path.shift(); 
                            } else {
                                enemy.x += (tdx/tdist) * moveSpeed;
                                enemy.y += (tdy/tdist) * moveSpeed;
                            }
                        } else {
                            enemy.x += (dx/dist) * moveSpeed;
                            enemy.y += (dy/dist) * moveSpeed;
                        }
                    } 
                    else if (dist <= 0.4 && player.hitCooldown === 0) {
                        applyDamage("enemy_attack");
                        player.hitCooldown = 60; 
                        enemy.stunTimer = 40;    
                    }
                }
            }
        }

        function checkStageCollision() {
            const checkDist = 0.8;
            const checkX = Math.floor(player.x + player.dirX * checkDist);
            const checkY = Math.floor(player.y + player.dirY * checkDist);
            const tile = worldMap[checkY][checkX];
            
            if (tile >= 2 && tile <= 7) {
                activeStageNode = { x: checkX, y: checkY, type: tile };
                openStage(tile);
            }
        }

        function gameLoop() {
            updateGame();
            render3D();
            renderMinimap();
            requestAnimationFrame(gameLoop);
        }

        function openStage(tileId) {
            initAudio(); 
            isGamePaused = true;
            keys = {w:false, s:false, a:false, d:false}; 
            
            const stage = stagesData[tileId];
            document.getElementById('stage-title').innerText = stage.title;
            document.getElementById('stage-desc').innerText = stage.desc;
            
            const btnContainer = document.getElementById('btn-container');
            btnContainer.innerHTML = ''; 
            uiLayer.style.display = 'flex';

            if (tileId === 2) {
                createBtn("홀", () => checkAns("홀", Math.random() < 0.5 ? "홀" : "짝"));
                createBtn("짝", () => checkAns("짝", Math.random() < 0.5 ? "홀" : "짝"));
            } else if (tileId === 3) {
                createBtn("붉은 문", () => checkAns("붉은 문", Math.random() < 0.5 ? "붉은 문" : "검은 문"));
                createBtn("검은 문", () => checkAns("검은 문", Math.random() < 0.5 ? "붉은 문" : "검은 문"));
            } else if (tileId === 4) {
                const ans = Math.floor(Math.random() * 6) + 1;
                for(let i=1; i<=6; i++) createBtn(`${i}번 약실`, () => checkAns(i, ans));
            } else if (tileId === 5) {
                const ans = Math.floor(Math.random() * 3) + 1;
                createBtn("상자 A", () => checkAns(1, ans));
                createBtn("상자 B", () => checkAns(2, ans));
                createBtn("상자 C", () => checkAns(3, ans));
            } else if (tileId === 6) {
                createBtn("앞면", () => checkAns("앞면", Math.random() < 0.5 ? "앞면" : "뒷면"));
                createBtn("뒷면", () => checkAns("뒷면", Math.random() < 0.5 ? "앞면" : "뒷면"));
            } else if (tileId === 7) {
                const ans = Math.floor(Math.random() * 5) + 1;
                for(let i=1; i<=5; i++) createBtn(i, () => checkAns(i, ans));
            }
        }

        function createBtn(text, callback) {
            const btn = document.createElement('button');
            btn.innerText = text;
            btn.onclick = callback;
            document.getElementById('btn-container').appendChild(btn);
        }

        function checkAns(playerChoice, realAnswer) {
            uiLayer.style.display = 'none';
            stageCooldown = 60; 
            
            if (playerChoice === realAnswer) {
                showMsg("길이 열렸다.");
                worldMap[activeStageNode.y][activeStageNode.x] = 0; 
                clearedStages++;
                
                if(clearedStages >= 6) {
                    setTimeout(() => {
                        alert("탈출구다...!!\n\n-- YOU SURVIVED --");
                        location.reload();
                    }, 500);
                }
            } else {
                applyDamage("wrong_answer");
            }
            isGamePaused = false;
        }

        function applyDamage(reason) {
            player.hp--;
            
            const overlay = document.getElementById('damage-overlay');
            overlay.style.opacity = 1;
            setTimeout(() => overlay.style.opacity = 0, 300);

            playHorrorSound();

            if (player.hp <= 0) {
                render3D(); 
                setTimeout(() => {
                    alert("추격자에게 붙잡혀 어둠 속으로 끌려갔습니다...\n\n처음부터 다시 시작합니다.");
                    location.reload();
                }, 100);
                return;
            } 
            
            if (reason === "wrong_answer") {
                if (!enemy.active) {
                    enemy.active = true;
                    showMsg(`오답입니다... 어둠 속에서 무언가 깨어났습니다! (남은 기회: ${player.hp}번)`);
                } else {
                    showMsg(`오답입니다! (남은 기회: ${player.hp}번)`);
                }
            } else if (reason === "enemy_attack") {
                showMsg(`추격자에게 공격당했습니다! (남은 기회: ${player.hp}번)`);
            }
        }

        function showMsg(text) {
            const msgBox = document.getElementById('message');
            msgBox.innerText = text;
            msgBox.style.display = 'block';
            setTimeout(() => msgBox.style.display = 'none', 2500);
        }

        gameLoop();
    </script>
